syntax $('[attribute]')
example:
$('[title]') all elements that have title attribute
$('div[title]') all elements divs with title attribute
$('[title="someTitle"]') all elements that have title attribute with value someTitle
$('[title]').css('border', '5px solid red') search all element with title attribute and set border as 5px solid red

$('div[title]') all div elements that have title attribute
$('div[title][style]') all div elements that have title as well as style
$('div[title="a"][style="b"]') all div elements that have title = a as well as style=b
